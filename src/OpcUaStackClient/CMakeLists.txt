include_directories(
    ../inc
	${Boost_INCLUDE_DIRS}
)

)

install(FILES ${INC_OpcUaStackClient_SecureChannel} DESTINATION ${INC_DIR}/OpcUaStack/OpcUaStackClient/SecureChannel)
